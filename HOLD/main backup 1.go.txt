// ===============================================================================================

//	HELPER FUNCTIONS

// ===============================================================================================
func json2Struct(fp string, t any) {
	rawFile, err1 := os.Open(fp)
	if err1 != nil {
		fmt.Println(err1)
	}
	fmt.Println("Successfully Opened snipux-settings.json")

	defer rawFile.Close()

	rawFileByte, err2 := io.ReadAll(rawFile)
	if err2 != nil {
		fmt.Println(err2)
	}

	json.Unmarshal(rawFileByte, t)
}
